FROM alpine:3.8
#指定作者
#MAINTAINER aaa

#USER daemo

​

#EXPOSE 80

#设置环境变量
#ENV <key>=<value>
​
#拷贝本地文件到指定路径
#ADD <src>... <dest> 支持网络
#COPY <src>... <dest> 只支持本地
​
​
#构件容器时运行
#RUN ls
​
#启动时执行 同时存在cmd 和 ENTRYPOINT 将忽略 ENTRYPOINT 或 cmd 为参数
#ENTRYPOINT ["executable", "param1", "param2"]
#CMD ls -al

#挂载目录 目录自身挂载 无法指定

#功能为容器启动时要运行的命令
​

#ONBUILD [INSTRUCTION]
