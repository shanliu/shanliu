      //头文件
template <class T>class ttest{

ttest();
~ttest();

};

m_pT = new T[1];
}

delete [] m_pT ;
}
template <class T> void ttest<T>::sett(T t) {
m_pT[0] = t;
}
template <class T> T ttest<T>::gett() {
T t = m_pT[0];
return t;
}
//使用
ttest<int> tvar;
tvar.sett(1);
